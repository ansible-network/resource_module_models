# Examples using overridden state to set configuration
  # Before state:
    config:
      enabled: False
      polling_interval: 50
      collectors:
        - address: 1.1.1.1
          port: 6343
          network-instance: default 
      interfaces:
        - name: Ethernet0
          enabled: false
        - name: Ethernet8
          enabled: false
        - name: Ethernet16
          enabled: false
        - name: Ethernet24
          enabled: false

  # Example:
    - name: "override sets to passed in task"
      sonic_sFlow:
        config:
          enabled: True
          agent: Ethernet0
          interfaces:
            - name: Ethernet0
              enabled: True
        state: overridden

  # After state:
    config:
      enabled: True
      agent: Ethernet0
      interfaces:
        - name: Ethernet0
          enabled: true
