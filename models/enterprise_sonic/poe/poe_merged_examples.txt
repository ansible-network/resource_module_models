# Using merged to set poe global settings
  # Before state:
  # config:
  #   global:
  #     auto_reset: false

  # Example:
    - name: "add poe global settings"
      sonic_poe:
        config:
          global:
            auto_reset: true
            power_mgmt_model: 'class'
            usage_threshold: 300
        state: merged

  # After state:
  # config:
  #   global:
  #     auto_reset: true
  #     power_mgmt_model: 'class'
  #     usage_threshold: 300
  # ------
